910  mlhsInnerSubst<matlabtypesubstitute,newDataOut> smoothAndInterpolate(matlabtypesubstitute newDataIn) {
911 
912  /* add data */
913  obj.buffer1= [obj.buffer1; newDataIn];
914  idx = (obj.p1_Npre+1):(size(obj.buffer1,1)-obj.p1_Npost);
915  nBodyContent = numel(idx);
916 
917  /* terminate if no new body data */
918  if nBodyContent == 0
919  newDataOut = [];
920  return;
921  end
922 
923  /* for (any) content data, process
924  * n.b. this could probably all be vectorised, to speed things
925  * up considerably */
926  for i = idx
927  /* fill-in missing values by linear interpolation */
928  for xy = 1:2
929  k = 1;
930  while isnan(obj.buffer1(i,xy)) && k <= obj.p1_iw
931  obj.buffer1(i,xy) = obj.buffer1(i-1,xy)*(1-1/(k+1)) + obj.buffer1(i+k,xy)*(1/(k+1));
932  k = k+1;
933  end
934  end
935  end
936 
937  /* smooth using running mean to effect a low-pass filter in
938  * the time-domain */
939  tmp = obj.buffer1;
940  m = obj.p1_ww2;
941  for i = idx
942  xysmoothed = mean(tmp((i-m):(i+m),1:2),1);

944  obj.buffer1(i,1:2) = xysmoothed;
945  end

947  end

982 
983  /* identify saccades
984  *d = obj.buffer2(:,7); % euclidean distance in degrees visual angle */
985  v = obj.buffer2(:,8); /* velocity (deg/sec) */
986 
987  A = obj.buffer2(:,9); /* acceleration (deg/sec)^2 */
988 
989  distmetric_deg = sqrt(sum((obj.buffer2(idx-obj.p2_dOffsetPrev,1:2)-obj.buffer2(idx+obj.p2_dOffsetPost,1:2)).^2,2));
990  distmetric_deg = distmetric_deg / obj.pixel_per_dg; /* ivis.math.IvUnitHandler.getInstance().px2deg(distmetric_deg);% convert to degrees visual angle */
991

994  end
995

997  * for changes in acceleration greater than a given cutoff (default
998  * is a change in velocity > 100 degrees per second) */
999  isSaccadeOn = (A(idx) > obj.accelCriterion & v(idx) > obj.velocityCriterion & distmetric_deg > obj.distanceCriterion);
1000  isSaccadeOff = (A(idx) < -obj.accelCriterion & v(idx) > obj.velocityCriterion);
1001

1003  * coordinates
1004  * (transient missing data points will have been interpolated above) (probably a cleaner way of doing this) */

1006 
1007  isBlink = isBlink | any(diff(obj.buffer2(:,3)) > obj.blinkTemporalMin_secs); /* EyeX */
1008 
1009 
1010  /* set bit-flags */
1011  obj.buffer2(idx,6) = 0 + isSaccadeOn*2^(obj.SACCADE_ON_BITPOS-1) + isSaccadeOff*2^(obj.SACCADE_OFF_BITPOS-1) + isBlink*2^(obj.BLINK_BITPOS-1); /* ALT: bitset.m */
1012 
1013 
1014  /* % tmp?, store */

1016  obj.buffer2(idx,8) = v(idx);
1017  obj.buffer2(idx,9) = A(idx);
1018 
1019  /* update data store */
1020  newDataOut = obj.buffer2(idx,:);
1021  obj.buffer2(1:nBodyContent,:) = [];
1022  }
1038  mlhsInnerSubst<matlabtypesubstitute,newDataOut> postHocCleanEvents(matlabtypesubstitute newDataIn) {

1043  idx = (obj.p3_Npre+1):(size(obj.buffer3,1)-obj.p3_Npost);
1044  nBodyContent = numel(idx);
1045 
1046  /* terminate if no new body data */
1047  if nBodyContent == 0
1048  newDataOut = [];

1050  end
1051 
1052  /* for (any) content data, process [NB: cannot be vectorized as
1053  * relies on sequentially updated values] */
1054  for i = idx

1056  * or too close to a preceding/succeeding blink */
1057  if bitget(obj.buffer3(i,6), obj.SACCADE_ON_BITPOS)==1 /* change to switch if more conditions arise */
1058

1062  obj.buffer3(i,6) = bitset(obj.buffer3(i,6), obj.SACCADE_ON_BITPOS, 0);

1064  end
1065

1067  * this is currently a bit of a hack, and is intended for
1068  * TobiiEyeX style ('callback') updating rather than Tobii
1069  * style polling */
1070  if bitget(obj.buffer3(i,6), obj.BLINK_BITPOS)==1 /* change to switch if more conditions arise
1071  *if sum(bitget(obj.buffer3((i-obj.prevBlinkWindow):(i-1),6), obj.BLINK_BITPOS)==1) > 0 */
1072 
1073  if sum(bitget(obj.buffer3(1:(i-1),6), obj.BLINK_BITPOS)==1) > 0 /* important because if eyes closed may be accumulating ############# */
1074 
1075  obj.buffer3(i,6) = bitset(obj.buffer3(i,6), obj.BLINK_BITPOS, 0);
1076  end
1077  end
1078  end
