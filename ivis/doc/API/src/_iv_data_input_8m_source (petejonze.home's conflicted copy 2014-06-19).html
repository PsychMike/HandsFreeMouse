562  function newDataOut = smoothAndInterpolate(newDataIn) {
563 
564  /* add data */
565  obj.buffer1= [obj.buffer1; newDataIn];
566  idx = (obj.p1_Npre+1):(size(obj.buffer1,1)-obj.p1_Npost);
567  nBodyContent = numel(idx);
568 
569  /* terminate if no new body data */
570  if nBodyContent == 0
571  newDataOut = [];
572  return;
573  end
574 
575  /* for (any) content data, process
576  * n.b. this could probably all be vectorised, to speed things
577  * up considerably */
578  for i = idx
579  /* fill-in missing values by linear interpolation */
580  for xy = 1:2
581  k = 1;
582  while isnan(obj.buffer1(i,xy)) && k <= obj.p1_iw
583  /* buffer1(i,xy) = (buffer1(i-1,xy) + buffer1(i+k,xy)) / 2; */
584  obj.buffer1(i,xy) = obj.buffer1(i-1,xy)*(1-1/(k+1)) + obj.buffer1(i+k,xy)*(1/(k+1));
585  k = k+1;
586  end
587  end
588  end
589 
590  /* smooth using running mean to effect a low-pass filter in
591  * the time-domain */
592  tmp = obj.buffer1;
593  m = obj.p1_ww2;
594  for i = idx
595  smoothed = mean(tmp((i-m):(i+m),1),1);
596  obj.buffer1(i,~isnan(smoothed)) = smoothed;
597  end
598 
599  /* update data store, return any items that 'overflow' the
600  * buffer */
601  newDataOut = obj.buffer1(idx,:);
602  obj.buffer1(1:nBodyContent,:) = [];
603  }
617  function newDataOut = tagEvents(newDataIn) {
618 
619  /* add data */
620  obj.buffer2= [obj.buffer2; newDataIn];
621  idx = (obj.p2_Npre+1):(size(obj.buffer2,1)-obj.p2_Npost);
622  nBodyContent = numel(idx);
623 
624  /* terminate if no new body data */
625  if nBodyContent == 0
626  newDataOut = [];
627  return;
628  end
629 
630  /* identify saccades */
631  d = [NaN; sqrt(sum(diff(obj.buffer2(:,1:2),1).^2,2))]; /* calc euchlidean distance in points */
632  d = d / obj.pixel_per_dg; /* ivis.math.IvUnitHandler.getInstance().px2deg(d);% convert to degrees visual angle */
633  dt = [NaN; diff(obj.buffer2(:,3))]; /* time diff in seconds */
634  v = d./dt; /* calc velocity (deg/sec) */
635  A = [NaN; diff(v)]; /* calc acceleration */
636  distmetric = sqrt(sum((obj.buffer2(idx-obj.p2_dOffsetPrev,1:2)-obj.buffer2(idx+obj.p2_dOffsetPost,1:2)).^2,2));
637  distmetric = distmetric / obj.pixel_per_dg; /* ivis.math.IvUnitHandler.getInstance().px2deg(distmetric);% convert to degrees visual angle */
638 
639  /* calculate the times when (any) saccades occured by looking
640  * for changes in acceleration greater than a given cutoff (default
641  * is a change in velocity > 100 degrees per second) */
642  isSaccadeOn = (A(idx) > obj.accelCriterion & v(idx) > obj.velocityCriterion & distmetric > obj.distanceCriterion);
643  isSaccadeOff = (A(idx) < -obj.accelCriterion & v(idx) > obj.velocityCriterion);
644 
645  /* identify blink as any instance with a nan in both X and Y
646  * coordinates
647  * (transient missing data points will have been interpolated above) (probably a cleaner way of doing this) */
648  isBlink = all(isnan(obj.buffer2(idx,1:2)),2);
649 
650  /* set */
651  obj.buffer2(idx,6) = 0 + isSaccadeOn*obj.SACCADE_ON_CODE + isSaccadeOff*obj.SACCADE_OFF_CODE + isBlink*obj.BLINK_CODE;
652 
653  /* % tmp?, store */
654  obj.buffer2(idx,7) = distmetric;
655  obj.buffer2(idx,8) = v(idx);
656  obj.buffer2(idx,9) = A(idx);
657 
658  /* update data store */
659  newDataOut = obj.buffer2(idx,:);
660  obj.buffer2(1:nBodyContent,:) = [];
661  }
