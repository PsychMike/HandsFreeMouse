243  function [xPos , yPos , speed , potentialEnergy , kineticEnergy ] = calcNextTimestep() {
244  for i=1:obj.N
245  for j=1:obj.N /* set force/energobj.y interactions for each particle */
246  if i~=j /* don't take self inducting terms into account */
247 
248  /* compute the distance */
249  rij = sqrt((obj.x(i)-obj.x(j))^2+ (obj.y(i)-obj.y(j))^2);
250 
251  /* compute unit normal vector, n(1)=n_x, n(2)=n_y */
252  n = [ obj.x(i)-obj.x(j) obj.y(i)-obj.y(j) ]/norm([ obj.x(i)-obj.x(j) obj.y(i)-obj.y(j) ]);
253 
254  /* for x direction */
255  obj.f_x(j) = n(1)*-4*obj.epsilon*((6*obj.sigma^6)/rij^7 - (12*obj.sigma^12)/rij^13);
256 
257  /* for y direction */
258  obj.f_y(j) = n(2)*-4*obj.epsilon*((6*obj.sigma^6)/rij^7 - (12*obj.sigma^12)/rij^13);
259 
260  /* potential energy */
261  obj.V_j(j) = 4*obj.epsilon*((obj.sigma/rij)^12-(obj.sigma/rij)^6);
262 
263  else /* if i=j: force=0, potential=0 */
264  obj.f_x(j) = 0;
265  obj.f_y(j) = 0;
266  obj.V_j(j) = 0;
267  end
268  end
269 
270  /* Compute the position and velocitobj.y using Euler or Verlet */
271  oldX = obj.x;
272  oldY = obj.y;
273 
274  /* Forward Euler x */
275  obj.x(i) = obj.x(i) + obj.v(i)*obj.dt; /* overwrite old position with new */
276 
277  obj.v(i) = obj.v(i) + sum(obj.f_x)/obj.m(i)*obj.dt;
278 obj.v(i) = max(obj.v(i), -obj.maxVelocity);
279 obj.v(i) = min(obj.v(i), obj.maxVelocity);
280 
281 
282  /* Forward Euler y */
283  obj.y(i) = obj.y(i) + obj.u(i)*obj.dt; /* overwrite */
284  obj.u(i) = obj.u(i) + sum(obj.f_y)/obj.m(i)*obj.dt;
285 obj.u(i) = max(obj.u(i), -obj.maxVelocity);
286 obj.u(i) = min(obj.u(i), obj.maxVelocity);
287 
288 
289  /* If borders are on (i.e. not empty) then bounce-off if
290  * necessary */
291  if ~isempty(obj.borders)
292  if (obj.x(i) < obj.borders(1)+obj.sigma/4) || (obj.x(i) > obj.borders(2)-obj.sigma/4) /* obj.x */
293  obj.x(i) = 2*oldX(i) - obj.x(i);
294  obj.v(i) = -obj.v(i);
295  end
296  if (obj.y(i) < obj.borders(3)+obj.sigma/4) || (obj.y(i) > obj.borders(4)-obj.sigma/4) /* obj.y */
297  obj.y(i) = 2*oldY(i) - obj.y(i);
298  obj.u(i) = -obj.u(i);
299  end
300  end
301 
302 
303  /* Energy calculations */
304  obj.V(i) = sum(obj.V_j); /* Potential energy: sum of influence of other particles */
305  obj.T(i) = 1/2*obj.m(i)*(obj.v(i)^2+obj.u(i)^2); /* Kinetic energy: 1/2mv^2 */
306  end
307 
308  /* Return variables */
309  xPos = obj.x;
310  yPos = obj.y;
311  speed = sqrt(obj.v.^2 + obj.u.^2);
312  potentialEnergy = obj.V;
313  kineticEnergy = obj.T;
